@startuml
class UI {
  + main()
}

class MessageBoardFacade {
  + authenticate(expertId: UInt32): bool
  + getAssignedApplications(expertId: UInt32): List<GrantApplication>
  + startExpertiseReview(applicationId: UInt32): GrantApplication
  + submitExpertiseResult(applicationId: UInt32, result: ExpertiseResult): void
  + createGrantApplication(applicantId: UInt32): GrantApplication
  + attachDocuments(applicationId: UInt32, docs: List<File>): void
  + getGeneralStatistics(filters: StatisticsFilters): Statistics
  + getExpertStatistics(expertId: UInt32): ExpertStatistics
  + generateReport(filters: StatisticsFilters, format: ReportFormat): ReportFile
}

interface IUsersRepository {
  + FindById(id: uInt32): User
  + Create(user: User): User
  + Update(user: User): User
  + Delete(user: User)
}

interface IExpertiseRepository {
  + FindById(id: uInt32): GrantApplication
  + Create(expertId: UInt32): List<GrantApplication>
  + Update(applicationId: UInt32, result: ExpertiseResult): void
  + Delete(applicationId: UInt32): void
}

interface IGrantApplicationRepository {
  + findById(applicationId: UInt32): GrantApplication
  + create(application: GrantApplication): GrantApplication
  + addDocuments(applicationId: UInt32, docs: List<File>): void
  + Update(order: GrantApplication): Order
  + Delete(order: GrantApplication)
}

interface IStatisticsRepository {
  + getSummary(): String
  + calculateApprovalRate(): Float
  + calculateRejectionRate(): Float
  + calculateAverageFunding(): Float
  + updateStatistics(data: List<GrantApplication>): void
}

interface IApplicantRepository {
  + findById(applicantId: UInt32): Applicant
}

interface IExpertiseAdapter {
  + submitForExpertise(applicationId: UInt32): ExpertiseResult
  + finalizeApplicationReview(applicationId: UInt32, verdict: Verdict): void
}

class UsersRepository implements IUsersRepository{
  - database: Database
}

class ExpertiseRepository implements IExpertiseRepository {
  - database: Database
}

class GrantApplicationRepository implements IGrantApplicationRepository {
  - database: Database
}

class ApplicantRepository implements IApplicantRepository {
  - database: Database
}

class StatisticsRepository implements IStatisticsRepository {
  - database: Database
}

class Statistics {
  - id: UInt32
  - statistics: UInt32
}

class ExpertiseServiceAdapter implements IExpertiseAdapter {
  + submitForExpertise(applicationId: UInt32): ExpertiseResult
  + finalizeApplicationReview(applicationId: UInt32, verdict: Verdict): void
}

class NotificationService {
  + sendSMS(userId: uInt32, text: String)
  + sendEmail(userId: uInt32, text: String)
  + notifyDeliveryStatus(orderId: uInt32, status: String)
  + notifyModerationStatus(advertisementId: uInt32, status: String)
}

class Database {
  + query(sql: String): ResultSet
  + execute(command: String)
}

class User {
  + userId: uInt32
  + name: String
  + email: String
  + phoneNumber: String
  + location: String
  + profilePicture: ByteArray
}

class GrantApplication {
  - applicationId: UInt32
  - applicantId: UInt32
  - status: String
  - documents: List<File>
}

class Applicant {
  - applicantId: UInt32
  - name: String
  - contactInfo: String
}

class ExpertiseReview {
  - reviewId: UInt32
  - applicationId: UInt32
  - expertId: UInt32
  - verdict: Verdict
  - score: Float
  - comments: String
  - reviewDate: Date
}

class File {
  - fileId: UInt32
  - fileName: String
  - fileType: String
  - fileData: Byte[]
}
